git clone путь-к-файлу - клонирует, добавляет репозиторий в выбранную папку

git status - показывает статус (состаяние) ткущего локального репозитория

git add имя-файла - добавляет новый файл под контроль Git (при изменения файла, ему будет передан статус modifined - изменён)

git comit -am "комментарий к комиту"
